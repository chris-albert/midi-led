MIDI Note to Color

 - 0 -> show
 - 1 -> morph
 - 2 -> off
 - 3 -> red
 - 4 -> green
 - 5 -> blue
 - 6 -> white
 - 7 through 127 -> every hsl degree from 0 to 360 in steps of 3
